設備發送狀態流程圖

    stateDiagram-v2
        direction LR %% =================================
        %% 樣式定義 (關鍵部分，用於復刻原圖風格)
        %% =================================
        %% 定義藍色狀態框樣式：天藍色填充、黑色邊框、圓角
        classDef blueState fill:#89CFF0,stroke:#333,stroke-width:1.5px,color:black,rx:5,ry:5,font-weight:500;
        %% 定義灰色群組框樣式：淺灰色填充、深灰邊框、圓角
        classDef greyGroup fill:#E0E0E0,stroke:#777,stroke-width:1.5px,color:black,rx:8,ry:8,font-weight:bold;

        %% =================================
        %% 狀態與流程定義
        %% =================================
        
        %% 初始狀態指向 Idle
        [*] --> Idle

        %% 主路徑
        Idle --> Processing : add_command()
/ Process Next
        
        %% Processing 與 Timeout 的交互
        Processing --> Timeout : Outer Timeout
Exceeded
        Timeout --> Processing : Retry/Next
        
        %% 進入 Sending 複合狀態
        Processing --> Sending : _raw_send()

        %% 定義 Sending 複合狀態 (灰色大框)
        state Sending {
            direction TB [*] --> WaitingForData
            
            WaitingForData --> MatchingToken : Data Received
            MatchingToken --> WaitingForData : Token Not Matched
            
            MatchingToken --> TailStage : Token Matched
            
            %% 自我循環
            TailStage --> TailStage : Data Received
(Reset Timer)
        }

        %% 從 Sending 內部狀態流出
        TailStage --> CommandFinished : Settle Timer
Timeout
        
        %% CommandFinished 的分支
        CommandFinished --> Idle : queue_finished
        CommandFinished --> Timeout : Next Command

        %% =================================
        %% 應用樣式
        %% =================================
        %% 將藍色樣式應用於所有普通狀態
        class Idle,Processing,Timeout,CommandFinished blueState
        class WaitingForData,MatchingToken,TailStage blueState
        %% 將灰色樣式應用於 Sending 群組
        class Sending greyGroup
